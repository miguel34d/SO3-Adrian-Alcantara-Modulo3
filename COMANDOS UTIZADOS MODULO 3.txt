#!/bin/bash
# Script para crear un backup comprimido de /home/tu_usuario

# Obtener fecha y hora actual en formato dd-mm-yyyy:hh:mm
FECHA=$(date +"%d-%m-%Y:%H:%M")

# Usuario actual
USUARIO=$(whoami)

# Nombre del archivo
ARCHIVO="backup-${USUARIO}-${FECHA}.tar.gz"

# Crear comprimido
tar -czvf /home/$USUARIO/$ARCHIVO /home/$USUARIO

echo "Backup creado correctamente: /home/$USUARIO/$ARCHIVO"

#!/bin/bash
# Script que guarda la salida de ifconfig en un archivo en el escritorio

# Preguntar nombre del archivo
echo -n "Introduce el nombre que deseas ponerle al archivo: "
read NOMBRE

# Usuario actual
USUARIO=$(whoami)

# Directorio Escritorio (Desktop)
ESCRITORIO="/home/$USUARIO/Desktop"

# Crear archivo con el nombre dado
ifconfig > "$ESCRITORIO/$NOMBRE.txt"

echo "Archivo creado correctamente en $ESCRITORIO/$NOMBRE.txt"
-------------------------------------------
sudo chmod +x ifconfig.sh
./ifconfig.sh
